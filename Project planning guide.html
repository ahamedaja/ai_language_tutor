from fastapi import HTTPException
import requests
import os

GEMINI_API_KEY = os.getenv("GEMINI_API_KEY")
GEMINI_URL = "https://api.gemini.example/v1/complete"

def call_gemini_for_evaluation(user_input, user_level, lesson_context):
    prompt = f"""
    You are an English tutor. Return JSON only with: errors, correction, explanations, score, suggested_exercises.
    user_level: {user_level}
    lesson_context: {lesson_context}
    user_input: \"\"\"{user_input}\"\"\"
    """

    if resp.status_code != 200:
        raise HTTPException(status_code=502, detail="AI service error")
    return resp.json()  # expect structured JSON per template

from fastapi import FastAPI, Depends, HTTPException
from pydantic import BaseModel
from typing import List, Optional
from app import ai_service, db

app = FastAPI(title="AI Language Tutor API")


class ExerciseSubmit(BaseModel):
    lesson_id: int
    user_input: str
    mode: Optional[str] = "free_text"


class ErrorItem(BaseModel):
    span: str
    type: str
    explanation: str


class EvaluationResult(BaseModel):
    errors: List[ErrorItem]
    correction: str
    explanations: List[str]
    score: float
    suggested_exercises: List[str]


@app.post("/api/exercises/submit", response_model=EvaluationResult)
async def submit_exercise(payload: ExerciseSubmit, user=Depends(db.get_current_user)):
    # store raw input (optional)
    # call AI service
    ai_resp = await ai_service.evaluate_text(
        user_input=payload.user_input,
        user_level=user.level,
        lesson_context={"lesson_id": payload.lesson_id}
    )
    # ai_resp expected to be dict matching EvaluationResult
    return ai_resp

import os
import httpx
from typing import Dict, Any

GEMINI_API_KEY = os.getenv("GEMINI_API_KEY")
GEMINI_URL = os.getenv("GEMINI_URL", "https://api.gemini.example/v1/complete")

async def evaluate_text(user_input: str, user_level: str, lesson_context: Dict[str, Any]) -> Dict[str, Any]:
    prompt = build_prompt(user_input, user_level, lesson_context)
    headers = {"Authorization": f"Bearer {GEMINI_API_KEY}", "Content-Type": "application/json"}
    payload = {"prompt": prompt, "max_tokens": 400, "temperature": 0.2}
    async with httpx.AsyncClient(timeout=15) as client:
        resp = await client.post(GEMINI_URL, json=payload, headers=headers)
    resp.raise_for_status()
    # assume Gemini returns text which is JSON only; parse it
    data = resp.json()
    # Validate keys, fallback if necessary
    return data

def build_prompt(user_input: str, user_level: str, lesson_context: dict) -> str:
    return f"""
You are an English tutor. Return JSON ONLY with keys:
errors: [{{span, type, explanation}}], correction: string, explanations: [string], score: float, suggested_exercises: [string].
User level: {user_level}
Lesson context: {lesson_context}
User input: \"\"\"{user_input}\"\"\"
Be concise. Use short explanations (1-2 sentences max).
"""

frontend/src/pages/LessonView.tsx (simplified)

tsx
import React, { useState } from "react";
import axios from "axios";

export default function LessonView({ lessonId }: { lessonId: number }) {
  const [input, setInput] = useState("");
  const [feedback, setFeedback] = useState<any>(null);
  const submit = async () => {
    const res = await axios.post("/api/exercises/submit", { lesson_id: lessonId, user_input: input });
    setFeedback(res.data);
  };
  return (
    <div className="p-4">
      <h2 className="text-xl font-semibold">Lesson {lessonId}</h2>
      <textarea value={input} onChange={(e)=>setInput(e.target.value)} className="w-full h-32 border" />
      <div className="mt-2">
        <button onClick={submit} className="px-4 py-2 bg-blue-600 text-white rounded">Submit</button>
      </div>
      <div className="mt-4">
        {feedback && (
          <div className="p-4 border rounded">
            <h3 className="font-bold">Corrected</h3>
            <p className="text-green-700">{feedback.correction}</p>
            <h4 className="mt-2 font-bold">Explanations</h4>
            <ul>{feedback.explanations.map((e:string,i:number)=><li key={i}>{e}</li>)}</ul>
          </div>
        )}
      </div>
    </div>
  );
}

4) Gemini prompt templates and expected JSON schema

Below are strong prompt templates to get deterministic, structured outputs from Gemini. Use low temperature and instruct JSON only.

Prompt template (evaluation)

yaml
SYSTEM: You are a concise English tutor. Return JSON only.

INSTRUCTIONS:
- Analyze the user's sentence.
- Return a JSON object with keys: errors, correction, explanations, score, suggested_exercises.
- errors: list of { span, type, explanation } where span is the exact substring that is problematic.
- correction: corrected sentence or best rephrasing (one string).
- explanations: short strings (1-2 sentences max) explaining the mistakes and why correction works.
- score: float between 0.0 and 1.0 representing overall accuracy.
- suggested_exercises: list of short IDs for recommended follow-up exercises.

INPUT:
user_level: {user_level}
lesson_context: {lesson_context}
user_input: """{user_input}"""

Output only JSON. Do not add any other text.

Example filled prompt

python
user_level: intermediate
lesson_context: "past simple practice"
user_input: """He go to school yesterday."""

Expected JSON schema (strict)

json
{
  "errors": [
    {
      "span": "go",
      "type": "verb tense",
      "explanation": "Use past simple 'went' for completed past actions; 'go' is present tense."
    }
  ],
  "correction": "He went to school yesterday.",
  "explanations": [
    "Use the past simple for completed actions; change 'go' to 'went'."
  ],
  "score": 0.6,
  "suggested_exercises": ["past-simple-01", "regular-irregular-verbs-02"]
}

Notes and robust parsing strategy

  • In the backend, after receiving the model output, validate JSON using a schema (Pydantic) and fall back to heuristics if keys are missing.

  • Keep temperature low (0.00.3) to reduce hallucinations.

  • Add a wrapper instruction to the model to only return the JSON and nothing else; then parse safely.

  • Log raw model output for debugging and improvements.
